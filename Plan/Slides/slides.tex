\section{Project Organisation}

\frame{
	\frametitle{Project Tasks}

	\begin{block}{Task 1: VF2 Implementation}
	Implementation of the VF2 graph matching algorithm, contributing
	it to the JgraphT library.
	\end{block}

	\begin{block}{Task 2: Changes on ISGCI}
	Changing the way smallgraphs are stored in ISGCI from
	adjacency matrices to JgraphT graph objects.
	\end{block}

	\begin{block}{Task 3: Implementing the forbids-relation}
	Combining the results from task 1 and 2 to implement the
	smallgraph forbids-relation in ISGCI.
	\end{block}
}

\frame{
	\frametitle{Task parallelization}

	Since tasks 1 and 2 are independent, we have decided to split our team
	and work on both in parallel:

	\begin{columns}
	\column{0.45\textwidth}
	\begin{block}{Task 1: VF2 Algorithm}
	\begin{itemize}
	\item Forking JgraphT
	\item Implementing VF2
	\end{itemize}	

	\end{block}
	\column{0.45\textwidth}
	\begin{block}{Task 2: ISGCI Storage}
	\begin{itemize}
	\item Forking ISGCI
	\item Replacing graph storage 
	\end{itemize}
	\end{block}
	\end{columns}


	\begin{columns}
	\column{0.96\textwidth}
	\begin{block}{Task 3: Forbids Relation}
	\begin{itemize}
	\item Experts from both tasks work on the final task together.
	\item Utilizing our VF2 implementation and integrating it in ISGCI.
	\item Implementing the forbids-relation.
	\end{itemize}

	\end{block}
	\end{columns}
}

\subsection{Development Methodology}
\frame{
	\frametitle{Development Methodology}
	\begin{block}{Scrum as a Project Management Framework}
	\begin{itemize}
	\item Dynamic human-ressources allocation
	\item Low bureaucratic overhead
	\item Flexible customer contact
	\item Self-imposed quality assurance guarantees:
		\begin{itemize}
		\item Documentation quality
		\item Rigorous testing
		\item Extensive code reviews
		\end{itemize}
	\item Finer-grained methods will be choosen during the analysis phase,
		depending on:

		\begin{itemize}
		\item Project requirements
		\item Team member preferences
		\item Customer guidelines	
		\end{itemize}

	\end{block}
